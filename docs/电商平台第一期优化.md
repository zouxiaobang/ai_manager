# 电商平台第一期优化方案

## 一、优化概述

本文档针对 `admin-backend` 和 `admin-web` 电商模块进行系统性分析，从**业务功能、代码组织、性能、安全性、可维护性**五个维度提出优化建议，形成可执行的第一期优化路线图。

### 2.3 库存反向操作（高优先级）

**问题**：退款/取消订单时没有将库存加回，欠货记录没有核销接口。

**优化方案**：

```java
// EcSalesOrderInventorySupport 新增方法
public void reclaimSkuCode(Long orderId, Long orderLineId, String skuCode, int reclaimQty) {
    EcInventory inventory = ecInventoryMapper.selectOne(...);
    if (inventory != null) {
        inventory.setQuantity(inventory.getQuantity() + reclaimQty);
        ecInventoryMapper.updateById(inventory);
        // 记录日志
        EcInventoryLog log = new EcInventoryLog();
        log.setChangeType("RECLAIM");
        log.setChangeQty(reclaimQty);
        ecInventoryLogMapper.insert(log);
    }
    // 核销欠货记录
    clearShortage(orderId, orderLineId, reclaimQty);
}

public void clearShortage(Long orderId, Long orderLineId, int clearedQty) {
    EcSalesOrderShortage shortage = ecSalesOrderShortageMapper.selectOne(...);
    if (shortage != null) {
        shortage.setClearedQty(shortage.getClearedQty() + clearedQty);
        if (shortage.getClearedQty() >= shortage.getShortQty()) {
            shortage.setStatus("CLEARED");
        }
        ecSalesOrderShortageMapper.updateById(shortage);
    }
}
```

**收益**：
- 修复退款不还库存的业务缺陷
- 完善欠货核销流程

```java
@Data
@TableName("ec_express_weight_tier")
public class EcExpressWeightTier {
    private Long id;
    private Long priceId;      // 关联 ec_express_price
    private BigDecimal minWeight;  // 最小重量（kg）
    private BigDecimal maxWeight;  // 最大重量（kg）
    private BigDecimal price;      // 对应价格
}

// 计算逻辑改为动态遍历
public BigDecimal calculateExpressFee(Long stationId, BigDecimal weightKg, String provinceName) {
    EcExpressPrice price = findPrice(stationId, provinceName);
    List<EcExpressWeightTier> tiers = findTiers(price.getId());
    
    // 遍历档位找匹配
    for (EcExpressWeightTier tier : tiers) {
        if (weightKg.compareTo(tier.getMinWeight()) >= 0 
            && weightKg.compareTo(tier.getMaxWeight()) <= 0) {
            return tier.getPrice();
        }
    }
    // 超档处理
    return calculateOverTier(price, weightKg);
}
```

### 3.3 表单草稿自动保存

```typescript
// composables/useDraftForm.ts
export function useDraftForm<T>(key: string, defaultForm: T) {
  const form = reactive<T>({ ...defaultForm });
  const STORAGE_KEY = `draft_${key}`;

  // 加载草稿
  onMounted(() => {
    const saved = localStorage.getItem(STORAGE_KEY);
    if (saved) {
      Object.assign(form, JSON.parse(saved));
    }
  });

  // 自动保存
  watch(form, (newVal) => {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(newVal));
  }, { deep: true });

  // 清除草稿
  function clearDraft() {
    localStorage.removeItem(STORAGE_KEY);
    Object.assign(form, defaultForm);
  }

  return { form, clearDraft };
}
```

### 3.4 列表缓存与乐观更新

```typescript
// composables/useListCache.ts
export function useListCache<T>(cacheKey: string) {
  const cache = ref<Map<string, T>>(new Map());

  function get(key: string): T | undefined {
    return cache.value.get(key);
  }

  function set(key: string, value: T) {
    cache.value.set(key, value);
  }

  function invalidate(key?: string) {
    if (key) {
      cache.value.delete(key);
    } else {
      cache.value.clear();
    }
  }

  return { cache, get, set, invalidate };
}
```
